SELECT Sales.*, Client.Name AS Client, BusinessUnit.Name AS BusinessUnit,AppUser.Id AS AppUserId,AppUser.UserName AS SalesManager FROM Sales INNER JOIN Client ON Sales.ClientId = Client.Id INNER JOIN BusinessUnit ON Sales.BusinessUnitId = BusinessUnit.Id INNER JOIN AppUser ON Sales.SalesManagerId = AppUser.Id INNER JOIN UserRole ON AppUser.Id = UserRole.UserId INNER JOIN Role ON UserRole.RoleId = Role.Id WHERE Role.Name = 'Sales Manager' AND 1=1 AND Sales.SalesManagerId in (30) AND Sales.Deleted = 0 ORDER BY Sales.Id Desc
